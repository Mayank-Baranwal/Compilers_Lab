        | stmt

stmt := SELECT LT condition GT LP Table_Name RP
        | PROJECT LT attribute_list GT LP Table_Name RP
        | LP Table_Name GP CARTESIAN_PRODUCT LP Table_Name GP
        | LP Table_Name GP EQUI_JOIN LT condition GT LP Table_Name GP

condition := temp_cond OR condition 
            | temp_cond

temp_cond := expression AND temp_cond
            | expression

expression := column op STRING
            | STRING op column
            | column op column
            | column op INT
            | INT op column
            | INT op INT
            | STRING op STRING
            | LP condition RP
            | NOT condition   

op := LT
    | GT
    | LTE
    | GTE
    | EQUAL
    | NOT_EQUAL

attribute_list := column COMMA attribute_list
                | column

column := column_name 
        | Table_Name DOT column_name
